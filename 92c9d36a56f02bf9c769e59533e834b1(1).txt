validate_recommended_position_detailed <- function(df) {
  
  valid_categories <- c("Strong Long", "Mild Long", "Neutral / Caution", 
                       "Mild Risk-off", "Strong Risk-off", 
                       "High Uncertainty - Neutral", "Cautious Strong", "Cautious Mild")
  
  original_count <- nrow(df)
  
  df_clean <- df %>%
    mutate(
      recommended_position = coalesce(recommended_position, "Neutral / Caution"),
      recommended_position = if_else(recommended_position %in% valid_categories, 
                                     recommended_position, "Neutral / Caution")
    )
  
  # Detailed Summary Table
  summary_table <- df_clean %>%
    count(recommended_position, name = "count") %>%
    mutate(percentage = round(count / sum(count) * 100, 2),
           flag = if_else(str_detect(recommended_position, "Cautious|Risk-off|High Uncertainty"), 
                          "High Caution", "Normal"))

  )
  
  cat("=== VALIDATION REPORT ===\n")
  print(summary_table)
  print(issues)
  
  return(list(validated_data = df_clean, summary = summary_table, issues = issues))
}

real_time_streamer <- function(update_interval = 60,   # seconds
                               assets = c("BTC-USD", "ETH-USD", "SPY", "AAPL"),
                               duration_hours = 8) {
  
  cat("🚀 Starting Real-Time V5.0 Streaming Pipeline...\n")

  }
  
  cat("✅ Real-time streaming session completed.\n")
  return(live_signals)
}

get_real_time_prices <- function(assets) {
  map_dfr(assets, ~ {
    tryCatch({
      getQuote(.x) %>%
        mutate(Asset = .x, Date = Sys.time(), Return = (Last - Open)/Open)
    }, error = function(e) tibble(Asset = .x, Date = Sys.time(), Return = NA_real_))
  })
}

# BINANCE WEBSOCKET INTEGRATION
# Real-time Price Feed for V5.0 Dashboard

library(tidyverse)
library(websocket)
library(jsonlite)
library(lubridate)

# Global reactive environment
binance_ws <- NULL
price_cache <- reactiveValues(data = tibble())

# Initialize Binance WebSocket
init_binance_websocket <- function(symbols = c("btcusdt", "ethusdt", "spy", "aapl"),
                                   on_message_callback = NULL) {
  
  # Convert to Binance format
  streams <- paste0(tolower(symbols), "@ticker")
  ws_url <- paste0("wss://stream.binance.com:9443/stream?streams=", 
                   paste(streams, collapse = "/"))
  
  binance_ws <<- WebSocket$`new(ws_url)
  
  binance_ws`$onMessage(function(event) {
    msg <- fromJSON(event$`data)
    
    if (!is.null(msg`$data)) {
      ticker <- msg$`data
      
      new_row <- tibble(
        Date = Sys.time(),
        Asset = str_to_upper(str_remove(ticker`$s, "USDT")),
        Price = as.numeric(ticker$`c),
        Return = as.numeric(ticker`$P) / 100,   # Price change percent
        Volume = as.numeric(ticker$`v)
      )
      
      # Update cache
      price_cache`$data <- bind_rows(price_cache$`data, new_row) %>%
        group_by(Asset) %>%
        slice_tail(n = 100) %>%   # Keep last 100 ticks per asset
        ungroup()
      
      # Trigger callback if provided
      if (!is.null(on_message_callback)) {
        on_message_callback(new_row)
      }
    }
  })
  
  binance_ws`$onOpen(function(event) {
    cat("✅ Binance WebSocket Connected\n")
  })
  
  binance_ws$`onClose(function(event) {
    cat("⚠️ Binance WebSocket Closed\n")
  })
  
  cat("🚀 Binance WebSocket initialized for symbols:", paste(symbols, collapse = ", "), "\n")
  return(binance_ws)
}

# Get latest cached prices
get_live_prices <- function() {
  price_cache`$data %>%
    group_by(Asset) %>%
    slice_tail(n = 1) %>%
    ungroup()
}

# Close connection
close_binance_websocket <- function() {
  if (!is.null(binance_ws)) {
    binance_ws$`close()
    cat("Binance WebSocket closed.\n")
  }
}

server <- function(input, output, session) {

  observe({
