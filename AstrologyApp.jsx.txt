function ZodiacWheel({ size = 180 }) {
  const signs = ["♈","♉","♊","♋","♌","♍","♎","♏","♐","♑","♒","♓"];
  const r = size/2, inner = r*0.62, mid = r*0.78;
  return (
    <svg width={size} height={size} style={{ opacity:0.35 }}>
      <circle cx={r} cy={r} r={r-4} fill="none" stroke={S.gold} strokeWidth={1} />
      <circle cx={r} cy={r} r={inner} fill="none" stroke={S.cardBorder} strokeWidth={1} />
      {signs.map((s,i) => {
        const a = (i*30 - 90) * Math.PI / 180;
        const sx = r + Math.cos(a) * mid, sy = r + Math.sin(a) * mid;
        const x1 = r + Math.cos(a) * inner, y1 = r + Math.sin(a) * inner;
        const x2 = r + Math.cos(a) * (r-4), y2 = r + Math.sin(a) * (r-4);
        return (
          <g key={i}>
            <line x1={x1} y1={y1} x2={x2} y2={y2} stroke={S.cardBorder} strokeWidth={0.5} />
            <text x={sx} y={sy} textAnchor="middle" dominantBaseline="central" fontSize={11} fill={S.gold}>{s}</text>
          </g>
        );
      })}
    </svg>
  );
}
