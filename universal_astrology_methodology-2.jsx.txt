function toLong(sign, deg, min=0) {
  const base = SIGN_BASE[sign];
  return base != null ? base + parseFloat(deg||0) + parseFloat(min||0)/60 : null;
}

function angleDiff(a, b) {
  let d = Math.abs(a - b) % 360;
  return d > 180 ? 360 - d : d;
}

function longToSignDeg(lon) {
  const l = ((parseFloat(lon) % 360) + 360) % 360;
  const sign = Object.entries(SIGN_BASE).reverse().find(([,base]) => l >= base)?.[0] || "Aries";
  return { sign, degree: (l - SIGN_BASE[sign]).toFixed(2) };
}

function calcAspects(bodies) {
  const results = [];
  const keys = Object.keys(bodies);
  for (let i = 0; i < keys.length; i++) {
    for (let j = i + 1; j < keys.length; j++) {
      const a = bodies[keys[i]]; const b = bodies[keys[j]];
      if (a.longitude == null || b.longitude == null) continue;
      const diff = angleDiff(parseFloat(a.longitude), parseFloat(b.longitude));
      const isLight = (n) => n === "Sun" || n === "Moon";
      for (const asp of ASPECTS_DEF) {
        const maxOrb = (isLight(keys[i]) || isLight(keys[j])) ? asp.orb + 2 : asp.orb;
        const orb = Math.abs(diff - asp.angle);
        if (orb <= maxOrb) {
          results.push({
            bodyA:keys[i], bodyB:keys[j], aspect:asp.name, symbol:asp.symbol,
            angle:asp.angle, orb:orb.toFixed(2), type:asp.type,
            quality:asp.quality, color:asp.color,
          });
          break;
        }
      }
    }
  }
  return results.sort((a,b) => parseFloat(a.orb) - parseFloat(b.orb));
}

function calcMidpoints(bodies) {
  const results = [];
  const keys = Object.keys(bodies).filter(k => bodies[k].longitude != null);
  for (let i = 0; i < keys.length; i++) {
    for (let j = i + 1; j < keys.length; j++) {
      const a = parseFloat(bodies[keys[i]].longitude);
      const b = parseFloat(bodies[keys[j]].longitude);
      let mp = (a + b) / 2;
      if (Math.abs(a - b) > 180) mp = (mp + 180) % 360;
      mp = ((mp % 360) + 360) % 360;
      const { sign, degree } = longToSignDeg(mp);
      const activators = [];
      for (const key of keys) {
        if (key === keys[i] || key === keys[j]) continue;
        const bl = parseFloat(bodies[key].longitude);
        for (const contact of [0, 90, 180, 45, 135]) {
          const target = ((mp + contact) % 360 + 360) % 360;
          const diff = angleDiff(bl, target);
          if (diff <= 1.5) { activators.push({ body:key, type:contact === 0 ? "direct" : `${contact}°`, orb:diff.toFixed(2) }); break; }
        }
      }
      results.push({ pair:`${keys[i]}/${keys[j]}`, bodyA:keys[i], bodyB:keys[j], longitude:mp.toFixed(2), sign, degree, activators });
    }
  }
  return results.sort((a,b) => b.activators.length - a.activators.length);
}

function detectPatterns(bodies, aspects) {
  const patterns = [];
  const trines = aspects.filter(a => a.aspect === "Trine");
  const squares = aspects.filter(a => a.aspect === "Square");
  const oppositions = aspects.filter(a => a.aspect === "Opposition");
  const quincunxes = aspects.filter(a => a.aspect === "Quincunx");
  const sextiles = aspects.filter(a => a.aspect === "Sextile");

  // Grand Trine
  for (let i = 0; i < trines.length; i++) {
    for (let j = i+1; j < trines.length; j++) {
      const u = [...new Set([trines[i].bodyA, trines[i].bodyB, trines[j].bodyA, trines[j].bodyB])];
      if (u.length === 3 && trines.find(t => (t.bodyA===u[0]||t.bodyB===u[0]) && (t.bodyA===u[2]||t.bodyB===u[2]))) {
        const el = ELEMENTS_MAP[bodies[u[0]]?.sign] || "?";
        patterns.push({ type:"Grand Trine", bodies:u, element:el, description:`${u.join(" △ ")} — ease and flow in ${el} expression` });
      }
    }
  }

  // T-Square
  for (const opp of oppositions) {
    for (const sq1 of squares) {
      const apex = [sq1.bodyA, sq1.bodyB].find(b => b !== opp.bodyA && b !== opp.bodyB);
      const base = [sq1.bodyA, sq1.bodyB].find(b => b === opp.bodyA || b === opp.bodyB);
      const other = base === opp.bodyA ? opp.bodyB : opp.bodyA;
      if (apex && squares.find(s => (s.bodyA===apex||s.bodyB===apex) && (s.bodyA===other||s.bodyB===other))) {
        if (!patterns.find(p => p.type === "T-Square" && p.bodies.includes(apex))) {
          const mode = MODES_MAP[bodies[apex]?.sign] || "?";
          patterns.push({ type:"T-Square", bodies:[opp.bodyA, opp.bodyB, apex], mode, description:`${mode} T-Square — Apex: ${apex}. Tension between ${opp.bodyA} ☍ ${opp.bodyB}` });
        }
      }
    }
  }

  // Yod
  for (let i = 0; i < quincunxes.length; i++) {
    for (let j = i+1; j < quincunxes.length; j++) {
      const apex = [quincunxes[i].bodyA, quincunxes[i].bodyB, quincunxes[j].bodyA, quincunxes[j].bodyB].find(
        b => [quincunxes[i].bodyA, quincunxes[i].bodyB].includes(b) && [quincunxes[j].bodyA, quincunxes[j].bodyB].includes(b)
      );
      if (!apex) continue;
      const base1 = [quincunxes[i].bodyA, quincunxes[i].bodyB].find(b => b !== apex);
      const base2 = [quincunxes[j].bodyA, quincunxes[j].bodyB].find(b => b !== apex);
      if (base1 && base2 && sextiles.find(s => (s.bodyA===base1||s.bodyB===base1) && (s.bodyA===base2||s.bodyB===base2))) {
        patterns.push({ type:"Yod", bodies:[base1, base2, apex], description:`"Finger of God" · Apex: ${apex} · ${base1} ⚹ ${base2} · fated vocation point` });
      }
    }
  }

  // Grand Cross
  if (oppositions.length >= 2 && squares.length >= 4) {
    for (let i = 0; i < oppositions.length; i++) {
      for (let j = i+1; j < oppositions.length; j++) {
        const four = [...new Set([oppositions[i].bodyA, oppositions[i].bodyB, oppositions[j].bodyA, oppositions[j].bodyB])];
        if (four.length === 4) {
          const allSquared = four.every(b => {
            const others = four.filter(x => x !== b);
            return others.filter(o => squares.find(s => (s.bodyA===b||s.bodyB===b) && (s.bodyA===o||s.bodyB===o))).length >= 2;
          });
          if (allSquared && !patterns.find(p => p.type === "Grand Cross")) {
            const mode = MODES_MAP[bodies[four[0]]?.sign] || "?";
            patterns.push({ type:"Grand Cross", bodies:four, mode, description:`${mode} Grand Cross — maximum tension and drive across all four ${mode} areas` });
          }
        }
      }
    }
  }

  // Stellium
  const signGroups = {};
  const houseGroups = {};
  for (const [name, b] of Object.entries(bodies)) {
    if (!["Ascendant","MC","Descendant","IC"].includes(name) && b.sign) {
      if (!signGroups[b.sign]) signGroups[b.sign] = [];
      signGroups[b.sign].push(name);
    }
    if (!["Ascendant","MC","Descendant","IC"].includes(name) && b.house) {
      if (!houseGroups[b.house]) houseGroups[b.house] = [];
      houseGroups[b.house].push(name);
    }
  }
  for (const [sign, members] of Object.entries(signGroups)) {
    if (members.length >= 3) patterns.push({ type:"Stellium", bodies:members, sign, description:`${members.length} planets in ${sign}: ${members.join(", ")}` });
  }
  for (const [house, members] of Object.entries(houseGroups)) {
    if (members.length >= 3 && !patterns.find(p => p.type === "Stellium" && p.bodies.every(b => members.includes(b)))) {
      patterns.push({ type:"Stellium (House)", bodies:members, sign:`H${house}`, description:`${members.length} planets in House ${house}: ${members.join(", ")}` });
    }
  }

  return patterns;
}

function calcDignity(planet, sign) {
  const d = DIGNITIES_DATA[planet];
  if (!d) return { status:"Peregrine", score:0 };
  if (d.domicile?.includes(sign)) return { status:"Domicile",   score:5 };
  if (d.exaltation === sign)       return { status:"Exaltation", score:4 };
  if (d.detriment?.includes(sign)) return { status:"Detriment",  score:-3 };
  if (d.fall === sign)             return { status:"Fall",        score:-4 };
  return { status:"Peregrine", score:0 };
}

  const sun  = bodies["Sun"]?.longitude  != null ? parseFloat(bodies["Sun"].longitude)  : null;
  const moon = bodies["Moon"]?.longitude != null ? parseFloat(bodies["Moon"].longitude) : null;
  const asc  = bodies["Ascendant"]?.longitude != null ? parseFloat(bodies["Ascendant"].longitude) : null;
  if (sun == null || moon == null || asc == null) return parts;

  const sunHouse = bodies["Sun"]?.house || 0;
  const isDay = sunHouse >= 7 && sunHouse <= 12;
  const norm = (v) => ((v % 360) + 360) % 360;

  const add = (name, lon, formula, glyph) => {
    lon = norm(lon);
    const { sign, degree } = longToSignDeg(lon);
    parts[name] = { longitude:lon.toFixed(2), sign, degree, formula, glyph };
  };

  add("Part of Fortune", isDay ? asc+moon-sun : asc+sun-moon, isDay ? "ASC + ☽ − ☉" : "ASC + ☉ − ☽", "⊕");
  add("Part of Spirit",  isDay ? asc+sun-moon : asc+moon-sun, isDay ? "ASC + ☉ − ☽" : "ASC + ☽ − ☉", "⊗");

  const venus  = bodies["Venus"]?.longitude  != null ? parseFloat(bodies["Venus"].longitude)  : null;
  const mars   = bodies["Mars"]?.longitude   != null ? parseFloat(bodies["Mars"].longitude)   : null;
  const jup    = bodies["Jupiter"]?.longitude!= null ? parseFloat(bodies["Jupiter"].longitude): null;
  const mer    = bodies["Mercury"]?.longitude!= null ? parseFloat(bodies["Mercury"].longitude): null;
  const sat    = bodies["Saturn"]?.longitude != null ? parseFloat(bodies["Saturn"].longitude) : null;
  const mc     = bodies["MC"]?.longitude     != null ? parseFloat(bodies["MC"].longitude)     : null;
  const h8     = bodies["House 8"]?.longitude!= null ? parseFloat(bodies["House 8"].longitude): null;

  if (venus != null) add("Part of Eros",     isDay ? asc+venus-sun  : asc+sun-venus,  "ASC + ♀ − ☉ (Day)", "♡");
  if (mars  != null) add("Part of Courage",  isDay ? asc+mars-sun   : asc+sun-mars,   "ASC + ♂ − ☉ (Day)", "⚔");
  if (jup   != null && mer != null) add("Part of Wealth", asc+jup-mer, "ASC + ♃ − ☿", "💰");
  if (sat   != null) add("Part of Karma",    isDay ? asc+sat-sun    : asc+sun-sat,    "ASC + ♄ − ☉ (Day)", "♄");
  if (mc    != null) add("Part of Vocation", isDay ? mc+moon-sun    : mc+sun-moon,    "MC + ☽ − ☉ (Day)",  "⊙");
  if (venus != null && sat != null) {
    add("Part of Marriage", isDay ? asc+venus-sat : asc+sat-venus, isDay ? "ASC + ♀ − ♄ (♂)" : "ASC + ♄ − ♀ (♀)", "⚭");
  }
  if (h8 != null) add("Part of Death", isDay ? asc+h8-moon : asc+moon-h8, isDay ? "ASC + H8 − ☽" : "ASC + ☽ − H8", "✝");

  return parts;
}

function analyzeBalance(bodies) {
  const elements = { Fire:0, Earth:0, Air:0, Water:0 };
  const modes = { Cardinal:0, Fixed:0, Mutable:0 };
  const hemisNS = { North:0, South:0 };
  const hemisEW = { East:0, West:0 };
  const core = ["Sun","Moon","Mercury","Venus","Mars","Jupiter","Saturn","Uranus","Neptune","Pluto"];
  for (const [name, b] of Object.entries(bodies)) {
    if (!core.includes(name) || !b.sign) continue;
    if (ELEMENTS_MAP[b.sign]) elements[ELEMENTS_MAP[b.sign]]++;
    if (MODES_MAP[b.sign])    modes[MODES_MAP[b.sign]]++;
    if (b.house) {
      const h = parseInt(b.house);
      if (h >= 1 && h <= 6)  hemisNS.North++; else hemisNS.South++;
      if (h >= 10 || h <= 3) hemisEW.East++;  else hemisEW.West++;
    }
  }
  return { elements, modes, hemisNS, hemisEW };
}
