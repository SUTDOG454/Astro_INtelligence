/* ─── CALCULATION UTILITIES ──────────────────────────────────────────── */
const SIGN_BASE = {
  Aries:0, Taurus:30, Gemini:60, Cancer:90, Leo:120, Virgo:150,
  Libra:180, Scorpio:210, Sagittarius:240, Capricorn:270, Aquarius:300, Pisces:330,
};
const toLong = (sign, deg, min=0, sec=0) =>
  (SIGN_BASE[sign] ?? 0) + parseFloat(deg) + parseFloat(min)/60 + parseFloat(sec)/3600;

const angleDiff = (a, b) => { const d = Math.abs(a-b)%360; return d>180?360-d:d; };

const longToSignDeg = (lon) => {
  const l = ((parseFloat(lon)%360)+360)%360;
  const sign = Object.entries(SIGN_BASE).reverse().find(([,b])=>l>=b)?.[0]||"Aries";
  return { sign, degree: (l - SIGN_BASE[sign]).toFixed(2) };
};

const midpointLon = (a,b) => {
  const diff = Math.abs(a-b);
  let mp = diff>180 ? ((a+b)/2+180)%360 : (a+b)/2;
  return ((mp%360)+360)%360;
};

const getHouse = (lon, cusps) => {
  const l = ((parseFloat(lon)%360)+360)%360;
  for (let i=0; i<12; i++) {
    const start = cusps[i];
    const end   = cusps[(i+1)%12];
    if (start <= end) { if (l >= start && l < end) return i+1; }
    else              { if (l >= start || l < end)  return i+1; }
  }
  return 1;
};

/* ─── CALCULATION FUNCTIONS ─────────────────────────────────────────── */
function calcInterAspects(pbA, pbB) {
  const results = [];
  const coreA = ["Sun","Moon","Mercury","Venus","Mars","Jupiter","Saturn","Uranus","Neptune","Pluto","Chiron","TrueNode","Ascendant","MC","Vertex","Juno","Eros"];
  const coreB = ["Sun","Moon","Mercury","Venus","Mars","Jupiter","Saturn","Uranus","Neptune","Pluto","Chiron","TrueNode","Ascendant","MC","Vertex","Juno","Eros"];
  for (const ka of coreA) {
    if (pbA[ka] == null) continue;
    for (const kb of coreB) {
      if (pbB[kb] == null) continue;
      const diff = angleDiff(parseFloat(pbA[ka]), parseFloat(pbB[kb]));
      const isLum = ["Sun","Moon","Ascendant"].includes(ka) || ["Sun","Moon","Ascendant"].includes(kb);
      for (const asp of ASPECTS_DEF) {
        const maxOrb = asp.orb + (isLum ? 2 : 0);
        const orb = Math.abs(diff - asp.angle);
        if (orb <= maxOrb) {
          const key = `${ka}-${kb}-${asp.name}`;
          const revKey = `${kb}-${ka}-${asp.name}`;
          const weight = SYNASTRY_WEIGHTS[key] || SYNASTRY_WEIGHTS[revKey];
          results.push({
            bodyA:ka, bodyB:kb, aspect:asp.name, sym:asp.sym,
            angle:asp.angle, orb:orb.toFixed(2), type:asp.type,
            quality:asp.quality, color:asp.color,
            score: weight?.score || 0, cat: weight?.cat || "Neutral",
            label: weight?.label || "",
          });
          break;
        }
      }
    }
  }
  return results.sort((a,b) => parseFloat(a.orb)-parseFloat(b.orb));
}

function calcOverlays(pbB, hCuspsA) {
  const overlays = {};
  const planetOrder = ["Sun","Moon","Mercury","Venus","Mars","Jupiter","Saturn","Uranus","Neptune","Pluto","Chiron","TrueNode","Ascendant","MC"];
  for (const p of planetOrder) {
    if (pbB[p] == null) continue;
    const h = getHouse(pbB[p], hCuspsA);
    if (!overlays[h]) overlays[h] = [];
    overlays[h].push(p);
  }
  return overlays;
}

function calcComposite(pbA, pbB) {
  const comp = {};
  const keys = ["Sun","Moon","Mercury","Venus","Mars","Jupiter","Saturn","Uranus","Neptune","Pluto","Chiron","TrueNode","Ascendant","MC"];
  for (const k of keys) {
    if (pbA[k]!=null && pbB[k]!=null) {
      comp[k] = midpointLon(parseFloat(pbA[k]), parseFloat(pbB[k]));
    }
  }
  return comp;
}

function scoreCompatibility(aspects) {
  let total = 0, byCategory = {};
  for (const asp of aspects) {
    if (asp.score !== 0) {
      total += asp.score;
      byCategory[asp.cat] = (byCategory[asp.cat]||0) + asp.score;
    }
  }
  return { total, byCategory };
}
